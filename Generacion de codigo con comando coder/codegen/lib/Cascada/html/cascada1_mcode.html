1%--------------------------------------------------
  2% Funcion que agrupa los polos de un sistema
  3% racional en pares conjugados, obteniendo los
  4% coeficientes reales de su desarrollo en cascada.
  5% Llamada
  6%   [N,D,g]=cascada(B,A,agrupo)
  7%
  8% Parametros de entrada:
  9%     Como en el caso de RESIDUE.
 10%   B(M): coeficientes del numerador.
 11%   A(N): Coeficientes del denominador.
 12%   agrupo: si vale 1 tambien agrupo los coeficentes
 13%       reales en factores de orden 2.
 14%       Si no quiero que los agrupe le asigno
 15%       el valor 0.
 16%       NOTA: su valor por defecto es 1.
 17% Parametros de salida: 
 18%   Una expresion vale mas que mil palabras.
 19% 
 20%       B(z)        N(k,1)+N(k,2)z^(-1)+N(k,3)z^(-2)
 21% H(z)=------=g*MUL---------------------------------
 22%       A(z)        D(k,1)+D(k,2)z^(-1)+D(k,3)z^(-2)
 23%
 24%-----------------------------------------------------
 25function [N,D,g]=cascada(B,A,agrupo)
 26
 27% Se comprueba si se ha definido el valor de agrupo.
 28% En caso contrario se iguala a 1.
 29if (nargin==2),agrupo=1;end
 30
 31% Se comprueba que todos los coeficientes
 32% de A y B son reales.
 33if (sum(imag(B))~=0),
 34error('Los coeficientes de B tienen que ser reales');
 35end
 36if (sum(imag(A))~=0),
 37error('Los coeficientes de A tienen que ser reales');
 38end
 39
 40% Se calculan las raices de A y B, y su ganancia.
 41c=roots(B);
 42d=roots(A);
 43g=B(1);
 44
 45% La agrupacion se basa en que la funcion roots nos
 46% devuelve los polos complejos y sus polos conjugados
 47% en posiciones consecutivas.
 48
 49% En primer lugar se reordenan los polos y ceros
 50% para facilitar el posterior algoritmo.
 51% La ordenacion consiste en poner primero polos y ceros
 52% complejos y luego los reales.
 53% Si no quiero que me agrupe los reales, pongo el valor
 54% cero detras de cada polo o cero real.
 55c_cmpx=c(find(real(c)~=c));
 56d_cmpx=d(find(real(d)~=d));
 57if (~agrupo),
 58   c_real=zeros(2*(length(c)-length(c_cmpx)),1);
 59   if (length(c_real)~=0),
 60       c_real(1:2:length(c_real))=c(find(real(c)==c));
 61   end;
 62   d_real=zeros(2*(length(d)-length(d_cmpx)),1);
 63   if (length(d_real)~=0),
 64    d_real(1:2:length(d_real))=d(find(real(d)==d));
 65   end;
 66
 67else
 68    c_real=[];d_real=[];
 69    for k=1:length(c),
 70        if (real(c(k))==c(k)),
 71            c_real=[c_real c(k)];
 72        end;
 73    end;
 74    for k=1:length(d),
 75        if (real(d(k))==d(k)),
 76            d_real=[d_real d(k)];
 77        end;
 78    end;
 79%   c_real=c(find(real(c)==c));
 80%   d_real=d(find(real(d)==d));
 81end
 82
 83c1=[c_cmpx; c_real'];
 84d1=[d_cmpx; d_real'];
 85
 86% Si la longitud no es par se anado un cero al final.
 87if ((fix(length(c1)/2))~=(length(c1)/2)),c1=[c1;0];end;
 88if ((fix(length(d1)/2))~=(length(d1)/2)),d1=[d1;0];end;
 89
 90% Calculo su factores N.
 91for i=2:2:length(c1),
 92    N(i/2,:)=real(poly([c1(i-1) c1(i)]));
 93end;
 94
 95% Calculo su factores D.
 96for i=2:2:length(d1),
 97    D(i/2,:)=real(poly([d1(i-1) d1(i)]));
 98end;
 99
100return
